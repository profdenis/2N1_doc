                System.out.println("Vous avez choisi l'option 2 !");

                System.out.println("Vous avez choisi l'option 3 !");

                System.out.println("Vous voulez quitter !");
                return true;
            default:
                System.out.println("Option invalide. SVP choisir une option valide.");
        }
        return false;
    }
}
